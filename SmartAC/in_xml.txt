public void ReadXML(string name)
{
    var b = new Path { title = name }; //ищем файл с титлом, соотв. названию
    var writer = new System.Xml.Serialization.XmlSerializer(typeof(Path));
    var wfile = new System.IO.StreamWriter(@"c:\options.xml");
    writer.Serialize(wfile, b);
    wfile.Close();

    System.Xml.Serialization.XmlSerializer reader = 
        new System.Xml.Serialization.XmlSerializer(typeof(Path));
    System.IO.StreamReader file = new System.IO.StreamReader(@"c:\options.xml");
    Path overview =  (Path)reader.Deserialize(file);
    file.Close();

    return Path overview;

}
